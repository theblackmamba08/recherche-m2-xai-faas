def max_consecutive_zeros(s):
    """Return max run length of consecutive zeros in series s."""
    max_run = 0
    current = 0
    for v in s:
        if v == 0:
            current += 1
            max_run = max(max_run, current)
        else:
            current = 0
    return max_run
